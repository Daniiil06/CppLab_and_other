void privedenie(double **arr) {

	/*p = arr[0];
	arr[0] = arr[1];
	arr[1] = p;*/

	for (int j = 0; j < n; j++) {
		max = arr[j][j];
		for (int i = 0; i < n; i++)
			if (arr[i][j] > max) {
				max = arr[i][j];
				p = arr[j];
				arr[j] = arr[i];
				arr[i] = p;
			}
	}



	int k = 0;
	for (int j = 0; j < n; j++) {
		for (int i = 0; i < n; i++) {
			if (arr[j][j] < arr[i][j]) k++;
		}
	}
	if (k > 1) cout << "\nFalse\n";
	else cout << "\nTrue\n";
}

			MaxDiff = fabs(Answr[i] - (arr[i][n] - sum) / arr[i][i]);
		cout << "\n" << (arr[i][n] - sum) / arr[i][i] << "";

	cout << "\n\n\n" << k << ": " << MaxDiff << "\n\n\n\n";
}
